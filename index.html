        // Utility function to extract the main category/artist name from the title

        function getCategoryName(title) {

            // Remove date, numbering, and common suffixes (Audio) to get the main group

            let cleanTitle = title

                .replace(/^\d{4}[ -]+\d{2}[ -]+\d{2}[ -]+/g, '') // Remove YYYY-MM-DD

                .replace(/^\d{4} - /, '') // Remove YYYY -

                .replace(/\s+\(Audio\)$/i, '') // Remove (Audio)

                .replace(/\s*-\s*[\d]{2}[-\s]*[^\s]+.*$/, '') // Remove track numbers and subsequent titles (e.g., ' 01-Zahořely')

                .replace(/:.*$/, '') // Remove everything after a colon (e.g., 'Josef Strubl: Český svaz žen...')

                .trim();

           

            // Special handling for common groups/events

            if (cleanTitle.includes("Hradišťan") || cleanTitle.includes("Wolfgang Saus")) return "Hradišťan a Přátelé";

            if (cleanTitle.includes("Spirituál kvintet")) return "Spirituál Kvintet";

            if (cleanTitle.includes("Traband") || cleanTitle.includes("Mirek Kemel")) return "Traband a Příbuzní";

            if (cleanTitle.includes("Jarda Svoboda")) return "Jarda Svoboda a Přátelé člověka";

            if (cleanTitle.includes("Vladimír Merta")) return "Vladimír Merta Trio/Hromosvod";

            if (cleanTitle.includes("Ulrychovi") || cleanTitle.includes("Javory")) return "Hana a Petr Ulrychovi a Javory";

            if (cleanTitle.includes("Galerie 12") || cleanTitle.includes("Lamr")) return "Galerie 12 / Vernisáže";

            if (cleanTitle.includes("Blondýna & Cicvárek")) return "Blondýna & Cicvárek";

            if (cleanTitle.includes("Michal Vaněk")) return "Michal Vaněk";

            if (cleanTitle.includes("Iva Bittová")) return "Iva Bittová a Spolupracovníci";

            if (cleanTitle.includes("Blue Effect") || cleanTitle.includes("Radim Hladík")) return "Radim Hladík a Blue Effect";


            // If it starts with a year and no clear artist, group as 'Ostatní'

            if (/^\d{4}/.test(cleanTitle)) {

                let parts = cleanTitle.split(' ');

                // Use the first 2-3 words as a likely artist name if not too long

                if (parts.length > 1 && parts[0].length < 10) {

                    cleanTitle = parts.slice(0, 2).join(' ');

                }

            }


            // Fallback for general historical/event videos

            if (cleanTitle.includes("Náměšť nad Oslavou") || cleanTitle.includes("O poklad Anežky České") || cleanTitle.includes("Zbraslav")) {

                return "Historické/Místní Události";

            }

            if (cleanTitle.includes("Prezentace Folkových prázdnin") || cleanTitle.includes("Klip FP")) {

                return "Folkové Prázdniny - Prezentace";

            }

           

            // Default: Use the cleaned title

            return cleanTitle || "Ostatní (Neznámý Interpret)";

        }


        // Pre-process data and group by category

        function groupData(data) {

            return data.reduce((acc, item) => {

                // Try to extract date for sorting (YYYY-MM-DD or YYYY)

                const dateMatch = item.title.match(/(\d{4}[ -]\d{2}[ -]\d{2}|\d{4})/)

                const dateRaw = dateMatch ? dateMatch[0].replace(/ /g, '-').trim() : '0000-00-00';

               

                // Add fields required for List.js sorting

                item.dateRaw = dateRaw;

                item.listTitle = item.title.replace(dateRaw, '').trim(); // Title without date prefix for A-Z sort

                item.listDate = dateRaw; // Formatted date for display


                const categoryName = getCategoryName(item.title);


                if (!acc[categoryName]) {

                    acc[categoryName] = []; // Inicializace pole pro novou kategorii

                }


                acc[categoryName].push(item);

                return acc;

            }, {});

        }

        // Funkce pro obsluhu globálního vyhledávání napříč všemi List.js instancemi

        function handleSearch(e) {

            const query = e.target.value.toLowerCase();

            const categoryItems = document.querySelectorAll('.category-item');


            categoryItems.forEach(categoryEl => {

                // Extrahovat název kategorie z hlavičky

                const categoryNameMatch = categoryEl.querySelector('.category-name').textContent.match(/(.*) \(\d+\)/);

                if (!categoryNameMatch) return; // Ochrana proti chybě

                const categoryName = categoryNameMatch[1].trim();

                const list = listInstances[categoryName];


                if (list) {

                    list.search(query);

                   

                    // Logika pro skrytí kategorie, pokud jsou všechny její položky odfiltrovány

                    const allItems = list.items.length;

                    const visibleItems = list.matchingItems.length;


                    if (visibleItems === 0 && allItems > 0) {

                        categoryEl.classList.add('hidden');

                    } else {

                        categoryEl.classList.remove('hidden');

                        // Rozbalit kategorii, pokud je v ní viditelná položka a uživatel vyhledává

                        if (query.length > 0 && visibleItems > 0 && categoryEl.classList.contains('collapsed')) {

                             categoryEl.classList.remove('collapsed');

                        }

                    }

                }

            });

        }


        // Funkce pro obsluhu tlačítek pro řazení (pro jednotlivé seznamy videí)

        function setupSortButtons() {

            document.querySelectorAll('.sort-btn[data-sort]').forEach(button => {

                button.addEventListener('click', (e) => {

                    const sortKey = e.target.getAttribute('data-sort');

                    const currentOrder = e.target.getAttribute('data-order');

                    const newOrder = currentOrder === 'asc' ? 'desc' : 'asc';

                   

                    // Aktualizovat stav tlačítka

                    e.target.setAttribute('data-order', newOrder);

                   

                    if (sortKey === 'list-title') {

                        e.target.textContent = `Řadit videa (Název ${newOrder === 'asc' ? 'A-Z' : 'Z-A'})`;

                    } else if (sortKey === 'list-date-raw') {

                        e.target.textContent = `Řadit videa (Datum ${newOrder === 'desc' ? 'sestupně' : 'vzestupně'})`;

                    }

                   

                    // Použít řazení na všechny List.js instance

                    Object.values(listInstances).forEach(list => {

                        list.sort(sortKey, { order: newOrder });

                    });

                });

            });


            // Speciální tlačítko pro řazení samotných kategorií

            document.getElementById('sort-category-name-btn').addEventListener('click', (e) => {

                const currentOrder = e.target.getAttribute('data-order');

                const newOrder = currentOrder === 'asc' ? 'desc' : 'asc';

                e.target.setAttribute('data-order', newOrder);

                e.target.textContent = `Řadit skupiny (Název ${newOrder === 'asc' ? 'A-Z' : 'Z-A'})`;


                const container = document.getElementById('category-container');

                const categories = Array.from(container.children);

               

                categories.sort((a, b) => {

                    // Získáme název kategorie bez počtu položek (čísla v závorkách)

                    const nameA = a.querySelector('.category-name').textContent.match(/(.*) \(\d+\)/)[1].trim();

                    const nameB = b.querySelector('.category-name').textContent.match(/(.*) \(\d+\)/)[1].trim();

                   

                    if (newOrder === 'asc') {

                        return nameA.localeCompare(nameB, 'cs');

                    } else {

                        return nameB.localeCompare(nameA, 'cs');

                    }

                });


                // Přidat seřazené elementy zpět do kontejneru

                categories.forEach(category => container.appendChild(category));

            });

        }
